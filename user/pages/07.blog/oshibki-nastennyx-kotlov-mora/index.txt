  document.addEventListener('DOMContentLoaded', function () {
    const buttons = document.querySelectorAll('button[data-target]');
    const sections = document.querySelectorAll('.error-section');

    buttons.forEach(button => {
      button.addEventListener('click', function () {
        const target = this.getAttribute('data-target');

        // Скрыть все секции
        sections.forEach(section => section.classList.add('is-hidden'));

        // Показать нужную
        const targetSection = document.getElementById('error-' + target);
        if (targetSection) {
          targetSection.classList.remove('is-hidden');
        }

        // Обновить активную кнопку
        buttons.forEach(btn => btn.classList.remove('is-primary'));
        this.classList.add('is-primary');
      });
    });
  });
