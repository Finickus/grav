<!-- Исправленный JavaScript: поиск по таблицам -->
<script>
  document.addEventListener('DOMContentLoaded', function () {
    const input = document.getElementById('partSearchInput');
    const btn btn-primary = document.getElementById('searchButton');

    if (!input || !btn btn-primary) return;

    function searchParts() {
      const query = input.value.trim().toLowerCase();
      const tables = document.querySelectorAll('.spec-table');

      tables.forEach(table => {
        const rows = table.querySelectorAll('tbody tr');
        let hasVisibleRow = false;

        rows.forEach(row => {
          const cells = row.cells;
          if (!cells || cells.length < 3) return;

          const pos = (cells[0].textContent || '').toLowerCase();
          const code = (cells[1].textContent || '').toLowerCase();
          const name = (cells[2].textContent || '').toLowerCase();
          const fullText = pos + ' ' + code + ' ' + name;

          if (query && fullText.includes(query)) {
            row.style.display = '';
            row.style.backgroundColor = '#e6f7ff';
            hasVisibleRow = true;
          } else {
            row.style.display = 'none';
            row.style.backgroundColor = '';
          }
        });

        // Скрыть/показать заголовок и изображение секции
        const sectionTitle = table.closest('div').previousElementSibling;
        if (sectionTitle && sectionTitle.classList.contains('h3')) {
          sectionTitle.style.display = hasVisibleRow ? '' : 'none';
          const figure = sectionTitle.nextElementSibling;
          if (figure && figure.tagName === 'FIGURE') {
            figure.style.display = hasVisibleRow ? '' : 'none';
          }
        }
      });
    }

    btn btn-primary.addEventListener('click', searchParts);
    input.addEventListener('keyup', function (e) {
      if (e.key === 'Enter') searchParts();
    });
  });
</script>
