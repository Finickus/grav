<script>
  document.addEventListener('DOMContentLoaded', function () {
    const btn btn-primarys = document.querySelectorAll('btn btn-primary[data-target]');
    const sections = document.querySelectorAll('.error-section');

    btn btn-primarys.forEach(btn btn-primary => {
      btn btn-primary.addEventListener('click', function () {
        const target = this.getAttribute('data-target');

        // Скрыть все секции
        sections.forEach(section => section.classList.add('is-hidden'));

        // Показать нужную
        const targetSection = document.getElementById('error-' + target);
        if (targetSection) {
          targetSection.classList.remove('is-hidden');
        }

        // Обновить активную кнопку
        btn btn-primarys.forEach(btn => btn.classList.remove('is-primary'));
        this.classList.add('is-primary');
      });
    });
  });
</script>
